protected function makeDsn($param)
{
    $dsn = 'mysql:host=' . (isset($param['host']) ? $param['host'] : 'localhost');
    if (isset($param['port']) && $param['port'] !== 3306) {
        $dsn .= ';port=' . $param['port'];
    }
    $dsn .= ';dbname=' . $param['dbname'];
    return $dsn;
}
