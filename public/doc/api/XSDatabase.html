public function query1($sql)
{
    $sql = preg_replace('/ limit\s+(\d+)(?:\s*,\s*(\d+)|\s+offset\s+(\d+))?\s*$/i', '', $sql);
    $sql .= ' LIMIT 1';
    $res = $this->query($sql);
    return (is_array($res) && isset($res[0])) ? $res[0] : false;
}
