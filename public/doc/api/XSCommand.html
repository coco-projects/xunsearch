XSCommand 命令对象 是与服务端交互的最基本单位, 命令对象可自动转换为通讯字符串, 命令结构参见 C 代码中的 struct xs_cmd 定义, 头部长度为 8字节.

public function __construct($cmd, $arg1 = 0, $arg2 = 0, $buf = '', $buf1 = '')
{
    if (is_array($cmd)) {
        foreach ($cmd as $key => $value) {
            if ($key === 'arg' || property_exists($this, $key)) {
                $this->$key = $value;
            }
        }
    } else {
        $this->cmd = $cmd;
        $this->arg1 = $arg1;
        $this->arg2 = $arg2;
        $this->buf = $buf;
        $this->buf1 = $buf1;
    }
}
